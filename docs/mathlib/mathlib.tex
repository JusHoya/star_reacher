% \starversion is overwritten at build time by version.tex, which .latexmkrc
% regenerates from `git describe --always` so the PDF names its exact source
% revision without manual bookkeeping. The fallback keeps the document
% buildable outside a git checkout.
\providecommand{\starversion}{unknown}
\InputIfFileExists{version.tex}{}{}

% \testid marks a validation-evidence test identifier. scripts/lint_chapters.py
% greps chapter sources for this macro, so every identifier it wraps is
% machine-checked to exist in the test tree (the FR-29 accretion gate).
% \detokenize keeps underscores in test identifiers printable in text mode.
\newcommand{\testid}[1]{\texttt{\detokenize{#1}}}

% Semantic math helpers so the conventions ratified in Chapter 1 are applied
% uniformly by every chapter instead of being re-spelled ad hoc.

\newcommand{\mat}[1]{\bm{#1}}                    % matrix
\newcommand{\dcm}[2]{\mat{C}_{#1}^{#2}}          % DCM mapping frame #1 to frame #2
\newcommand{\norm}[1]{\left\lVert #1 \right\rVert}
\newcommand{\quatmul}{\otimes}                   % Hamilton quaternion product
\newcommand{\skewm}[1]{\left[#1\right]_{\times}} % skew-symmetric cross-product matrix
\newcommand{\quat}[1]{\bar{#1}}                  % quaternion
